# The Town — Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Let players unlock a "Town" view where they own multiple houses, each with its own fully-separate garden (robots, upgrades, grass, layout) while sharing a global coin/gem wallet.

**Architecture:** Keep the existing single-lawn simulation untouched; turn the globals (`grass`, `tiles`, `flowerColors`, `grassSpecies`, `robots`, `bees`) into references bound to the **active house's** storage. A new `state.town` tree holds per-house state and view mode. A new `js/town.js` renders the street view. Unviewed owned houses accrue coins via a cheap flat idle formula. New tile types (`FENCE`, `PATH`, `DRIVEWAY`, `PATIO`, `HOUSE_BUILDING`, `POOL`) plus per-house ASCII layouts enable multi-zone yards with gates.

**Tech Stack:** Vanilla JS, classic `<script>` load order, Canvas2D, `localStorage` saves, no build step.

**Design reference:** `docs/plans/2026-04-17-the-town-design.md`.

**Verification convention (per task):**
1. `for f in js/*.js; do node --check "$f"; done` — must be clean.
2. Load `http://localhost:8765/` in a fresh browser tab (run `python3 -m http.server 8765` once). Check the manual smoke-test steps listed in each task. Use DevTools console to inspect `state`.
3. Commit only after both pass.

Pseudocode:
```js
function loadGame() {
  let raw = localStorage.getItem(SAVE_KEY);
  if (!raw) {
    // migrate v2 if present
    const v2 = localStorage.getItem(SAVE_KEY_V2);
    if (v2) {
      const migrated = migrateV2ToV3(JSON.parse(v2));
      raw = JSON.stringify(migrated);
      localStorage.setItem(SAVE_KEY, raw);
      // keep v2 around for one release as a safety net; do NOT delete yet.

function migrateV2ToV3(v2) {
  // v2 stored grass/tiles/… at the root. Wrap them as the starter house.
  return {
    version: 3,
    coins: v2.coins, gems: v2.gems, /* …copy all other non-world fields… */
    town: {
      unlocked: false,
      activeHouseKey: 'starter',
      inTownView: false,
      houses: {
        starter: {
          owned: true,
          gridW: CFG.gridW,
          gridH: CFG.gridH,
          grass: v2.grass || [],
          tiles: v2.tiles || [],
          flowerColors: v2.flowerColors || [],
          grassSpecies: v2.grassSpecies || [],
          zones: new Array(CFG.gridW * CFG.gridH).fill(0),
          totalTilesMowed: v2.totalTilesMowed || 0,
          initialized: true,
          robots: v2.robots || [],
          bees: v2.bees || [],
        },
      },
    },
  };

function restoreTown(t) {
  if (!t) { ensureStarterHouse(); return; }
  state.town.unlocked = !!t.unlocked;
  state.town.activeHouseKey = t.activeHouseKey || 'starter';
  state.town.inTownView = false;
  state.town.houses = {};
  for (const [k, h] of Object.entries(t.houses || {})) {
    const fresh = makePerHouseState(k);
    fresh.grass        = new Float32Array(h.grass || []);
    fresh.tiles        = new Uint8Array(h.tiles || []);
    fresh.flowerColors = new Uint8Array(h.flowerColors || []);
    fresh.grassSpecies = new Uint8Array(h.grassSpecies || []);
    fresh.zones        = new Uint8Array(h.zones || []);
    fresh.totalTilesMowed = h.totalTilesMowed || 0;
    fresh.initialized  = !!h.initialized;
    fresh.robots       = (h.robots || []).map(r => ({ ...r, lastTargetCheck: 0, target: null, dragging: false }));
    fresh.bees         = h.bees || [];
    state.town.houses[k] = Object.assign({ owned: !!h.owned }, fresh);
  }

    state.town.houses.starter = { owned: true, ...makePerHouseState('starter') };

}
```

### Task 7: Zone flood-fill on layout load

**Files:**
- Modify: `js/world.js`

**Step 1 — implement flood fill after layout paint:**
```js
// Assigns each grass tile a zone id (1..n). Zones are separated by any
// OBSTACLE tile. Non-grass walkable tiles (PATH/DRIVEWAY/PATIO) count as
// transit but are not themselves assigned zones — they get zone 0.
function computeZones(h, def) {
  const { gridW: w, gridH: gh } = def;
  h.zones.fill(0);
  let next = 1;
  const queue = [];
  for (let sy = 0; sy < gh; sy++) {
    for (let sx = 0; sx < w; sx++) {
      const sk = sy * w + sx;
      if (h.tiles[sk] !== T.GRASS || h.zones[sk] !== 0) continue;
      const id = next++;
      queue.length = 0;
      queue.push(sx, sy);
      h.zones[sk] = id;
      while (queue.length) {
        const y = queue.pop(), x = queue.pop();
        for (const [dx, dy] of [[1,0],[-1,0],[0,1],[0,-1]]) {
          const nx = x + dx, ny = y + dy;
          if (nx < 0 || ny < 0 || nx >= w || ny >= gh) continue;
          const nk = ny * w + nx;
          if (h.tiles[nk] !== T.GRASS || h.zones[nk] !== 0) continue;
          h.zones[nk] = id;
          queue.push(nx, ny);
        }
      }
    }
  }
}
```

**Step 2 — add a robot's current zone and prefer same-zone targets.** Change `pickTarget(r)` so the candidate list is filtered by `zones[k] === robotZone`; if no candidate found after the existing attempts, fall back to the unfiltered pool:

```js
function pickTarget(r) {
  const ts = tileSize;
  const rcx = Math.floor(r.x / ts), rcy = Math.floor(r.y / ts);
  const rk = (rcy >= 0 && rcx >= 0) ? idx(rcx, rcy) : 0;
  const robotZone = zones ? zones[rk] : 0;

  // Two-pass: prefer same-zone grass first, then any grass.
  for (const sameZoneOnly of [true, false]) {
    let best = null, bestScore = -Infinity;
    for (let attempt = 0; attempt < CFG.targetScanAttempts; attempt++) {
      const x = Math.floor(Math.random() * activeHouseDef().gridW);
      const y = Math.floor(Math.random() * activeHouseDef().gridH);
      const k = idx(x, y);
      if (tiles[k] !== T.GRASS) continue;
      if (grass[k] < 0.1) continue;
      if (sameZoneOnly && robotZone && zones[k] !== robotZone) continue;
      const d = Math.hypot((x + 0.5) * ts - r.x, (y + 0.5) * ts - r.y);
      const score = grass[k] - d * 0.002;
      if (score > bestScore) { bestScore = score; best = { x, y }; }
    }
    if (best) { r.target = best; r.lastTargetCheck = 0; return; }
  }
  r.target = null;
}
```

```js
// Ticks idle income for every owned house except the one passed in.
function tickIdleHouses(dt, skipKey) {
  for (const [k, h] of Object.entries(state.town.houses)) {
    if (!h.owned || k === skipKey) continue;
    if (state.fuel <= 0) continue;   // fuel gate applies across the board
    const gain = houseIdleCoinsPerSec(k) * dt;
    if (gain <= 0) continue;
    state.coins += gain;
    state.totalEarnedAllTime += gain;
    state.totalEarnedThisRun += gain;
  }
}
```
